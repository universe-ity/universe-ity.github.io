Shown below is the Mathematica code used to generate the applets.

Clear["Global`*"]
lambda = 0.500;
w0 = 5;
zR = Pi*w0^2/lambda;
w[z_] := w0*Sqrt[1 + (z/zR)^2];
(*----------------------Anas's Additions: Added table to plot for comparison with Activity results-----------------*)
CloudDeploy[Column[{
   Style["Waist values", Bold],
   Plot[w[z], {z, -20*zR, 20*zR}, ImageSize -> Medium],
   Style[ ""], Style[ ""],
   Style["Values", Bold], Style["    z          w[z]", Bold],
   Grid[Table[{z, w[z]}, {z, -20*zR, 20*zR, 40*zR/100}]]
   }], Permissions -> "Public"]
(*----------------------------------------------------------------------------------------------------------*)


Int[z_] := 1*Sqrt[2/Pi]/w[z]*Exp[-2*x^2/w[z]^2];
CloudDeploy[
 Manipulate[
  Plot[Int[z], {x, -2*w[z], 2*w[z]}], {z, -20*zR, 20*zR, zR/4}],
 Permissions -> "Public"]
Int3D[z_] := 1*Sqrt[2/Pi]/w[z]*Exp[-2*(x^2 + y^2)/w[z]^2];

P [z_, x_] = 1/2*( 1 + Erf[Sqrt[2]*(x)/w[z]]);
CloudDeploy[
 Manipulate[Column[{
    (*Plot3D[Int3D[z],{x,-2*w[z],2*w[z]},{y,-2*w[z],2*w[z]},

    Style["Knife Edge Scan", Bold],

     PlotRange -> {{-125, 125}, Automatic}],

G[z_, x_] = ND[P [z, x], x];
H[z_, x_] = N[D[P [z, x], x]];
model[x_] = ampl*Evaluate[PDF[NormalDistribution[x0, waist/2], x]];
altModel[x_] := ampl* Exp[-2*(x - x0)^2/waist^2]

CloudDeploy[Manipulate[Column[{
    Style["Derivative of Knife Edge Scan", Bold],
    Plot[H[z, x], {x, -2*w[z], 2*w[z]}, ImageSize -> Medium],
    Style[" "],
    Style["Data", Bold], Style["x          Derivative", Bold],
    Grid[mydata =
      Table[{x, H [z, x]}, {x, -2*w[z], 2*w[z], 6*w[z]/100}] ],
    Style[" "],
    Style["Fit", Bold],
    FindFit[mydata,
     model[x], {{ampl, FindPeaks[mydata[[All, 2]]][[1, 2]]}, {x0,
       0}, {waist, 50}}, x],
    FindFit[mydata,
     altModel[
      x], {{ampl, FindPeaks[mydata[[All, 2]]][[1, 2]]}, {x0,
       0}, {waist, 50}}, x]
    }, Dividers -> {False, {-4 -> Blue}}], {z, -20*zR, 20*zR,
   40*zR/800, Appearance -> "Labeled"}],
 Permissions -> "Public"]
